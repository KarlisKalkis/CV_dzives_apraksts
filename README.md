    print('Negatīvs vērtējums')
elif atz == 4:
    print('Gandrīz vidēji')
elif atz == 5:
    print('Vidēji')
elif atz == 6 :
    print('Gandrīz labi')
elif atz == 7:
    print('Labi')
elif atz == 8:
    print('Ļoti labi')
elif atz == 9:
    print(' Teicami')

    print('Izcili')

    print('Ir ievadīts nederīga atzīme')
